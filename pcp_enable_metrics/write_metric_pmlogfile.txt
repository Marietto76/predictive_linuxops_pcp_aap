# pminfo -h localhost | grep filesys
hinv.nfilesys

filesys.freefiles

# pminfo -h localhost | grep ds389.cn.current
ds389.cn.currentconnectionsatmaxthreads
ds389.cn.currentconnections

#pminfo -h localhost | grep ds389.cn.ops
ds389.cn.opsinitiated
ds389.cn.opscompleted






2. agregar la métrica para que sea escrita en el archivo pmlogger y colocar en que periodo de tiempo debe ser leida, cada 30 sec, cada 1 min, cada 30min, etc.

vi /var/lib/pcp/config/pmlogger/config.default

log mandatory on 30 second {
    ds389
}


3. reiniciar pmlogger
systemctl restart pmlogger
